\section{Why These Strategies Matter: The Cognitive Challenge}

Late-band problems (AMC12 P21--25, COMC Part C) deliberately exceed normal cognitive processing capacity. They require managing 6--8 simultaneous constraints while evaluating multiple solution paths. Success depends not on raw intelligence but on systematic cognitive strategies that compensate for working memory limitations.

\begin{strategybox}
\textbf{The 70-20-10 Practice Rule}: Structure your practice portfolio with 70\% ``stretch zone'' problems (just beyond current ability), 20\% ``expansion'' problems (requiring new techniques), and 10\% ``impossible'' problems (building persistence).
\end{strategybox}

\section{Six-to-Seven-Week Implementation Schedule}

\textbf{Important}: Don't implement every strategy at once. Start with 3--4 core strategies that address your biggest weaknesses.

\subsection{Weeks 1--2: Foundation (Focus on Three-Pass Protocol)}
\begin{itemize}[leftmargin=*]
    \item Master the \textbf{Three-Pass Protocol}:
    \begin{enumerate}
        \item First pass (3--4 min): Tag most accessible of P21--25
        \item Per attempt (2:30--4:00): Commit to conversion, push to subgoal
        \item Endgame (5--7 min left): Verify solves, revisit one promising partial
    \end{enumerate}
    \item Build initial pattern library (8--10 problem signatures)
    \item Practice Answer-Choice Leverage on AMC12 P16--20
    \item Begin error tracking with three categories: Computational, Conceptual, Strategic
\end{itemize}
\textbf{Weekly Goal}: Solve 3--4 P16--20 problems correctly using time protocols

\subsection{Weeks 3--4: Integration (Add Metacognitive Monitoring)}
\begin{itemize}[leftmargin=*]
    \item Expand pattern library to 15--18 signatures
    \item Apply all five strategies to AMC12 P21--23
    \item Implement the \textbf{Four Questions} (when stuck):
    \begin{enumerate}
        \item ``What am I trying to find?'' $\rightarrow$ Restate goal
        \item ``What haven't I used?'' $\rightarrow$ Check constraints
        \item ``Is there an easier version?'' $\rightarrow$ Consider simplification
        \item ``What would make this trivial?'' $\rightarrow$ Work backwards
    \end{enumerate}
    \item \textbf{End of Week 4}: Take full-length practice AMC12 for baseline
\end{itemize}
\textbf{Weekly Goal}: Solve 1--2 P21--23 problems correctly

\subsection{Weeks 5--6: Advanced Execution (Contest-Specific Tactics)}
\begin{itemize}[leftmargin=*]
    \item Full AMC12 P21--25 problem sets with time limits
    \item COMC Part C proof practice (40 points: 4 problems × 10 points each)
    \item Practice partial credit strategies for COMC:
    \begin{itemize}
        \item Write clear approach statement
        \item Show key insight or lemma
        \item Verify base cases
        \item State what remains to be proven
    \end{itemize}
    \item Weekly challenge: One AMC12 P24--25 problem (20--30 min limit)
\end{itemize}
\textbf{Weekly Goal}: Solve 2--3 P21--25 problems correctly

\subsection{Week 7: Competition Simulation and Emergency Protocols}
\begin{itemize}[leftmargin=*]
    \item 2--3 full timed mock exams
    \item Practice emergency protocols when stuck:
    \begin{itemize}
        \item AMC12: Use answers to guide approach, eliminate 2+ options, test easy cases
        \item COMC: Write approach, show insight, verify base cases, state remaining work
    \end{itemize}
    \item Review entire pattern library
    \item Test-day preparation checklist
\end{itemize}
\textbf{Goal}: Consistent performance on P21--25

\section{Test Day Preparation}

\subsection{Physical and Mental Readiness}
\begin{itemize}[leftmargin=*]
    \item Sleep 8+ hours for two nights before
    \item Protein-rich breakfast, avoid sugar crash
    \item Arrive early with backup supplies (calculator, pencils, eraser)
    \item Review pattern library morning of test
    \item Practice deep breathing for anxiety management
\end{itemize}

\subsection{During the Contest}
\begin{itemize}[leftmargin=*]
    \item First 5 minutes: Scan all problems, mark by perceived difficulty
    \item Problems 1--20: Move quickly but accurately (45--50 min)
    \item Problems 21--25: Apply full strategy arsenal (20--25 min)
    \item Last 5 minutes: Verify high-confidence answers, make educated guesses
\end{itemize}

\section{Continuous Improvement}

\subsection{After Each Practice Session}
\begin{enumerate}
    \item Update strategy log with new entries
    \item Identify which strategies felt most natural
    \item Note any new problem signatures discovered
    \item Celebrate small wins: ``Recognized pattern faster,'' ``Made fewer computational errors''
\end{enumerate}

\subsection{Weekly Review Process}
\begin{enumerate}
    \item Analyze error patterns: Are mistakes clustered in one category?
    \item Refine strategy triggers: Which cues most reliably indicate each strategy?
    \item Prune redundant patterns: Combine similar signatures
    \item Set specific goals for next week based on weaknesses
\end{enumerate}

\section{Key Mindset Principles}

\begin{strategybox}
\begin{enumerate}
    \item \textbf{Problems are puzzles}: Approach with curiosity, not fear
    \item \textbf{Every attempt teaches}: Failed approaches reveal gaps
    \item \textbf{Speed follows recognition}: Pattern library is crucial
    \item \textbf{Progress over perfection}: Partial solutions have value (especially COMC)
    \item \textbf{Trust the process}: Consistent practice with these strategies yields results
    \item \textbf{The gap between P20 and P25 is technique, not intelligence}
\end{enumerate}
\end{strategybox}
