\section*{Problem Statement}

A real-valued function $f$ has the property that for all real numbers $a$ and $b,$

$$f(a + b) + f(a - b) = 2f(a) f(b).$$

Which one of the following cannot be the value of $f(1)?$

$$\textbf{(A) } 0 \qquad \textbf{(B) } 1 \qquad \textbf{(C) } -1 \qquad \textbf{(D) } 2 \qquad \textbf{(E) } -2$$

\begin{problemconfigbox}
\subsection*{A. Problem Classification}
\begin{itemize}[leftmargin=*]
    \item \textbf{Problem Type}: Cannot Be True (find infeasible value)
    \item \textbf{Competition Context}: AMC 12B 2023, Problem 22
    \item \textbf{Difficulty Band}: Late-band (P22) - Very challenging
    \item \textbf{Mathematical Domain}: Functional Equations with Algebra
    \item \textbf{Estimated Time}: 5-8 minutes (requires substitution technique and inequality reasoning)
\end{itemize}

\subsection*{B. Problem Structure Identification}
\begin{itemize}[leftmargin=*]
    \item \textbf{Given Information}: 
    \begin{itemize}
        \item Functional equation: $f(a + b) + f(a - b) = 2f(a) f(b)$ for all real $a, b$
        \item $f$ is real-valued
        \item Need to determine which value $f(1)$ \emph{cannot} take
    \end{itemize}
    \item \textbf{Constraints}: 
    \begin{itemize}
        \item Functional equation must hold for \emph{all} real $a$ and $b$
        \item This severely restricts possible functions
    \end{itemize}
    \item \textbf{Objective}: 
    \begin{itemize}
        \item Find which of the five choices is impossible for $f(1)$
        \item Equivalently: show four choices are possible, one is impossible
    \end{itemize}
    \item \textbf{Hidden Assumptions}: 
    \begin{itemize}
        \item Key insight: substitute specific values of $a$ and $b$ to derive constraints
        \item Start with $a = b = 0$ to find $f(0)$
        \item Then use $a = b$ or $b = 0$ to get more information
        \item The functional equation may be related to trigonometric identities
    \end{itemize}
    \item \textbf{Answer Choice Leverage}: 
    \begin{itemize}
        \item (A) $f(1) = 0$: zero function always works
        \item (B) $f(1) = 1$: suggests $f(x) = 1$ (constant) or $f(x) = \cos(0) = 1$
        \item (C) $f(1) = -1$: suggests $f(x) = -1$ or boundary case
        \item (D) $f(1) = 2$: larger value, suggests $f(x) = \cosh(x)$ or similar
        \item (E) $f(1) = -2$: negative and large, suspicious
    \end{itemize}
    \item \textbf{Proof Requirements}: Not a proof problem; find infeasible value
\end{itemize}

    \item \textbf{Notation}:

        \item Given: $f(a + b) + f(a - b) = 2f(a)f(b)$

        \item Substitute $a = b = 0$ to find $f(0)$
        \item Substitute $a = b$ to get recursion for $f(2a)$
        \item Use inequality $f(a)^2 \geq 0$ to bound $f(2a)$
        \item Apply to $a = 1/2$ to bound $f(1)$

\begin{techniquesbox}
\subsection*{A. High-Probability Techniques}
\begin{itemize}[leftmargin=*]
    \item \textbf{Primary Technique}: Substitution + Inequality Bounding
    \item \textbf{Recognition Cues}: 
    \begin{itemize}
        \item Functional equation holds for all $a, b$
        \item Need to find impossible value
        \item Answer choices include extremes
    \end{itemize}
    \item \textbf{Application - Complete Solution}: 
    \begin{enumerate}
        \item \textbf{Step 1: Find $f(0)$}
        \begin{itemize}
            \item Substitute $a = b = 0$ into functional equation:
            \[f(0 + 0) + f(0 - 0) = 2f(0)f(0)\]
            \[f(0) + f(0) = 2f(0)^2\]
            \[2f(0) = 2f(0)^2\]
            \[f(0) = f(0)^2\]
            \item This means $f(0)(f(0) - 1) = 0$
            \item Therefore: $f(0) = 0$ or $f(0) = 1$
        \end{itemize}
        
        \item \textbf{Step 2: Case 1 - If $f(0) = 0$}
        \begin{itemize}
            \item Substitute $b = 0$ into functional equation:
            \[f(a + 0) + f(a - 0) = 2f(a)f(0)\]
            \[f(a) + f(a) = 2f(a) \cdot 0\]
            \[2f(a) = 0\]
            \item Therefore: $f(a) = 0$ for all $a$
            \item In particular: $f(1) = 0$, so choice (A) is \textbf{possible}
        \end{itemize}
        
        \item \textbf{Step 3: Case 2 - If $f(0) = 1$}
        \begin{itemize}
            \item Substitute $a = b$ into functional equation:
            \[f(a + a) + f(a - a) = 2f(a)f(a)\]
            \[f(2a) + f(0) = 2f(a)^2\]
            \[f(2a) + 1 = 2f(a)^2\]
            \[f(2a) = 2f(a)^2 - 1\]
        \end{itemize}
        
        \item \textbf{Step 4: Derive lower bound}
        \begin{itemize}
            \item Since $f(a)^2 \geq 0$ for all real $a$:
            \[f(2a) = 2f(a)^2 - 1 \geq 2 \cdot 0 - 1 = -1\]
            \item Therefore: $f(2a) \geq -1$ for all $a$
        \end{itemize}
        
        \item \textbf{Step 5: Apply to $f(1)$}
        \begin{itemize}
            \item Set $2a = 1$, so $a = \frac{1}{2}$:
            \[f(1) = f(2 \cdot \frac{1}{2}) \geq -1\]
            \item Therefore: $f(1) \geq -1$
        \end{itemize}
        
        \item \textbf{Step 6: Conclusion}
        \begin{itemize}
            \item From Case 1: $f(1) = 0$ is possible (choice A)
            \item From Case 2: $f(1) \geq -1$
            \item This means $f(1) \in \{0, -1, 1, 2, \ldots\}$ are potentially possible
            \item But $f(1) = -2 < -1$ violates the bound from Case 2
            \item And $f(1) = -2$ is not covered by Case 1 (which gives only $f(1) = 0$)
            \item Therefore: $f(1) = -2$ is \textbf{impossible}
        \end{itemize}
    \end{enumerate}
\end{itemize}

\begin{solutionbox}
\subsection*{A. Step-by-Step Solution}
\begin{enumerate}[leftmargin=*]
    \item \textbf{Find $f(0)$}: 
    \begin{itemize}
        \item Substitute $a = b = 0$ into $f(a+b) + f(a-b) = 2f(a)f(b)$:
        \[f(0) + f(0) = 2f(0)f(0)\]
        \[2f(0) = 2f(0)^2\]
        \[f(0) = f(0)^2\]
        \[f(0)(f(0) - 1) = 0\]
        \item Therefore: $f(0) = 0$ or $f(0) = 1$
    \end{itemize}
    
    \item \textbf{Case 1: $f(0) = 0$}: 
    \begin{itemize}
        \item Substitute $b = 0$ into functional equation:
        \[f(a) + f(a) = 2f(a) \cdot 0\]
        \[2f(a) = 0\]
        \item Therefore: $f(a) = 0$ for all $a$
        \item In particular: $f(1) = 0$
        \item So choice (A) is possible
    \end{itemize}
    
    \item \textbf{Case 2: $f(0) = 1$}: 
    \begin{itemize}
        \item Substitute $a = b$ into functional equation:
        \[f(2a) + f(0) = 2f(a)^2\]
        \[f(2a) + 1 = 2f(a)^2\]
        \[f(2a) = 2f(a)^2 - 1\]
    \end{itemize}
    
    \item \textbf{Derive Lower Bound}: 
    \begin{itemize}
        \item For any real number $a$, we have $f(a)^2 \geq 0$
        \item Therefore:
        \[f(2a) = 2f(a)^2 - 1 \geq 2(0) - 1 = -1\]
        \item So: $f(2a) \geq -1$ for all $a$
    \end{itemize}
    
    \item \textbf{Apply to $f(1)$}: 
    \begin{itemize}
        \item Let $2a = 1$, which means $a = \frac{1}{2}$
        \item Then:
        \[f(1) = f\left(2 \cdot \frac{1}{2}\right) \geq -1\]
        \item Therefore: $f(1) \geq -1$
    \end{itemize}
    
    \item \textbf{Conclusion}: 
    \begin{itemize}
        \item From our analysis, $f(1) \geq -1$
        \item Choices (A) 0, (B) 1, (C) $-1$, and (D) 2 all satisfy $f(1) \geq -1$
        \item Choice (E) $-2 < -1$ violates this bound
        \item Therefore, $f(1) = -2$ is \textbf{impossible}
    \end{itemize}
\end{enumerate}

\begin{warningbox}
\textbf{Warning}: Common pitfalls:
\begin{enumerate}
    \item \textbf{Forgetting Case 1}: Don't forget that $f(0) = 0$ is possible, which gives $f \equiv 0$ and shows choice (A) is possible.
    \item \textbf{Wrong substitution}: When finding $f(2a)$, must substitute $a = b$ (not $b = 2a$). The equation $f(2a) + f(0) = 2f(a)^2$ comes from $f(a+a) + f(a-a) = 2f(a)f(a)$.
    \item \textbf{Inequality direction}: The bound is $f(2a) \geq -1$, not $f(2a) \leq -1$. Check: $2f(a)^2 - 1 \geq 0 - 1 = -1$.
    \item \textbf{Wrong application}: Must use $2a = 1$ (so $a = 1/2$), not $a = 1$. We derived bound on $f(2a)$, not on $f(a)$ directly.
    \item \textbf{Forgetting both cases}: In Case 1, $f(1) = 0$ exactly. In Case 2, $f(1) \geq -1$. Together, this still gives $f(1) \geq -1$ overall (since $0 \geq -1$).
\end{enumerate}
\end{warningbox}
