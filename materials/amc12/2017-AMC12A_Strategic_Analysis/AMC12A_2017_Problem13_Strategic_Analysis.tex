\section*{Problem Statement}
Driving at a constant speed, Sharon usually takes $180$ minutes to drive from her house to her mother's house. One day Sharon begins the drive at her usual speed, but after driving $\frac{1}{3}$ of the way, she hits a bad snowstorm and reduces her speed by $20$ miles per hour. This time the trip takes her a total of $276$ minutes. How many miles is the drive from Sharon's house to her mother's house?

\[\textbf{(A)}\ 132 \qquad\textbf{(B)}\ 135 \qquad\textbf{(C)}\ 138 \qquad\textbf{(D)}\ 141 \qquad\textbf{(E)}\ 144\]

\vspace{0.3cm}

\begin{roadmapbox}
\subsection*{A. Step-by-Step Solution Plan}

\textbf{Step 1: Set Up Variables and First Equation (1 minute)}
\begin{itemize}[leftmargin=*]
    \item Let $D$ = total distance in miles
    \item Let $v$ = usual speed in miles per minute
    \item Normal trip: $D = v \cdot 180$ (distance = rate × time)
    \item Therefore: $v = \frac{D}{180}$ miles per minute
    \item \textbf{Checkpoint}: Units are consistent (miles per minute × minutes = miles)
\end{itemize}

\textbf{Step 2: Set Up Second Equation for Modified Trip (1.5 minutes)}
\begin{itemize}[leftmargin=*]
    \item First segment: distance $\frac{D}{3}$, speed $v$, time $t_1 = \frac{D/3}{v} = \frac{D}{3v}$
    \item Speed reduction: 20 mph = $\frac{20}{60} = \frac{1}{3}$ miles per minute
    \item Storm speed: $v - \frac{1}{3}$ miles per minute
    \item Second segment: distance $\frac{2D}{3}$, speed $v - \frac{1}{3}$, time $t_2 = \frac{2D/3}{v - 1/3} = \frac{2D}{3v - 1}$
    \item Total time: $t_1 + t_2 = 276$ minutes
    \item \textbf{Checkpoint}: Time equation is $\frac{D}{3v} + \frac{2D}{3v-1} = 276$
\end{itemize}

\textbf{Step 3: Substitute and Simplify (1.5 minutes)}
\begin{itemize}[leftmargin=*]
    \item Substitute $v = \frac{D}{180}$ into time equation:
    \[\frac{D}{3 \cdot \frac{D}{180}} + \frac{2D}{3 \cdot \frac{D}{180} - 1} = 276\]
    
    \item Simplify first term: $\frac{D}{\frac{3D}{180}} = \frac{180D}{3D} = 60$
    
    \item Simplify second term: $\frac{2D}{\frac{3D}{180} - 1} = \frac{2D}{\frac{3D - 180}{180}} = \frac{360D}{3D - 180}$
    
    \item Time equation becomes: $60 + \frac{360D}{3D - 180} = 276$
    
    \item \textbf{Checkpoint}: First segment takes exactly 60 minutes (makes sense: $\frac{1}{3}$ of 180 min)
\end{itemize}

\textbf{Step 4: Solve for D (1.5 minutes)}
\begin{itemize}[leftmargin=*]
    \item Isolate fraction: $\frac{360D}{3D - 180} = 216$
    
    \item Cross-multiply: $360D = 216(3D - 180)$
    
    \item Expand: $360D = 648D - 38880$
    
    \item Rearrange: $360D - 648D = -38880$
    
    \item Simplify: $-288D = -38880$
    
    \item Solve: $D = \frac{38880}{288} = 135$ miles
    
    \item \textbf{Checkpoint}: Answer is 135, which matches choice (B)
\end{itemize}

\textbf{Step 5: Verify Solution (1 minute)}
\begin{itemize}[leftmargin=*]
    \item Check normal trip: $v = \frac{135}{180} = \frac{3}{4}$ mi/min = 45 mph
    \item Time: $\frac{135 \text{ miles}}{45 \text{ mph}} = 3$ hours = 180 minutes $\checkmark$
    
    \item Check modified trip:
    \begin{itemize}[leftmargin=*]
        \item First $45$ miles at 45 mph: $\frac{45}{45} = 1$ hour = 60 min $\checkmark$
        \item Storm speed: $45 - 20 = 25$ mph
        \item Last $90$ miles at 25 mph: $\frac{90}{25} = 3.6$ hours = 216 min $\checkmark$
        \item Total: $60 + 216 = 276$ minutes $\checkmark$
    \end{itemize}
    
    \item \textbf{Checkpoint}: All conditions satisfied $\checkmark$
\end{itemize}

\subsection*{C. Common Pitfalls and Avoidance}
\begin{itemize}[leftmargin=*]
    \item \textbf{Pitfall 1}: Forgetting to convert 20 mph to miles per minute
    \begin{itemize}[leftmargin=*]
        \item \textit{Error}: Using $v - 20$ when $v$ is in mi/min
        \item \textit{Consequence}: Wrong answer, won't match choices
        \item \textit{Prevention}: Always write units explicitly; convert 20 mph $\rightarrow \frac{1}{3}$ mi/min
    \end{itemize}
    
    \item \textbf{Pitfall 2}: Using $t = D \cdot r$ instead of $t = \frac{D}{r}$
    \begin{itemize}[leftmargin=*]
        \item \textit{Error}: Multiplying distance by speed to get time
        \item \textit{Consequence}: Nonsensical equation
        \item \textit{Prevention}: Remember: time = distance ÷ rate (goes up when speed goes down)
    \end{itemize}
    
    \item \textbf{Pitfall 3}: Arithmetic errors in cross-multiplication
    \begin{itemize}[leftmargin=*]
        \item \textit{Error}: $360D = 648D - 38880$ simplifies incorrectly
        \item \textit{Consequence}: Wrong value of $D$
        \item \textit{Prevention}: Double-check sign when moving terms; verify final answer by back-substitution
