    \item Product becomes: $(p + 2i)(p - 2i)(q + 2i)(q - 2i)(r + 2i)(r - 2i)$
    \item Rearrange: $[(p + 2i)(q + 2i)(r + 2i)][(p - 2i)(q - 2i)(r - 2i)]$
    \item Note: $(p - x)(q - x)(r - x) = -f(x)$ where $f(x) = x^3 + 2x^2 - x + 3$
    \item With $x = -2i$: $(p + 2i)(q + 2i)(r + 2i) = -f(-2i)$
    \item With $x = 2i$: $(p - 2i)(q - 2i)(r - 2i) = -f(2i)$
    \item Product: $[-f(-2i)][-f(2i)] = f(2i) \cdot f(-2i)$

\end{enumerate}

\subsection*{B. Alternative Techniques}

\textbf{Method 2: Direct Expansion with Vieta's Formulas (H16)}

\textbf{Application}:
\begin{enumerate}
    \item Expand $(p^2 + 4)(q^2 + 4)(r^2 + 4)$:
    \[= p^2q^2r^2 + 4(p^2q^2 + p^2r^2 + q^2r^2) + 16(p^2 + q^2 + r^2) + 64\]
    
    \item From Vieta's: $pqr = -3$, so $p^2q^2r^2 = 9$
    
    \item Find $p^2 + q^2 + r^2$:
    \[p^2 + q^2 + r^2 = (p + q + r)^2 - 2(pq + pr + qr) = (-2)^2 - 2(-1) = 4 + 2 = 6\]
    
    \item Find $p^2q^2 + p^2r^2 + q^2r^2$:
    \begin{align*}
    (pq + pr + qr)^2 &= p^2q^2 + p^2r^2 + q^2r^2 + 2pqr(p + q + r)\\
    (-1)^2 &= p^2q^2 + p^2r^2 + q^2r^2 + 2(-3)(-2)\\
    1 &= p^2q^2 + p^2r^2 + q^2r^2 - 12\\
    p^2q^2 + p^2r^2 + q^2r^2 &= -11
    \end{align*}
    
    \item Substitute:
    \[9 + 4(-11) + 16(6) + 64 = 9 - 44 + 96 + 64 = 125\]
\end{enumerate}

\textbf{Method 3: Reduction of Power (Advanced)}

\textbf{Key idea}: Since $p$ is a root, $p^3 + 2p^2 - p + 3 = 0$, so $p^3 = -2p^2 + p - 3$

\textbf{Application}:
\begin{enumerate}
    \item From $p^3 + 2p^2 - p + 3 = 0$:
    \[p^2(p + 2) = p - 3\]
    \[p^2 = \frac{p - 3}{p + 2}\]
    
    \item Therefore:
    \[p^2 + 4 = \frac{p - 3}{p + 2} + 4 = \frac{p - 3 + 4(p + 2)}{p + 2} = \frac{5p + 5}{p + 2} = 5 \cdot \frac{p + 1}{p + 2}\]
    
    \item Same for $q$ and $r$:
    \[(p^2 + 4)(q^2 + 4)(r^2 + 4) = 125 \cdot \frac{(p + 1)(q + 1)(r + 1)}{(p + 2)(q + 2)(r + 2)}\]
    
    \item Using $(p + c)(q + c)(r + c) = -f(-c)$:
    \[= 125 \cdot \frac{-f(-1)}{-f(-2)} = 125 \cdot \frac{f(-1)}{f(-2)}\]
    
    \item Evaluate:
    \begin{align*}
    f(-1) &= (-1)^3 + 2(-1)^2 - (-1) + 3 = -1 + 2 + 1 + 3 = 5\\
    f(-2) &= (-2)^3 + 2(-2)^2 - (-2) + 3 = -8 + 8 + 2 + 3 = 5
    \end{align*}
    
    \item Result: $125 \cdot \frac{5}{5} = 125$
\end{enumerate}

\textbf{Method 4: Modular Arithmetic Check (Quick Verification)}

From expansion: $(pqr)^2 + 4(\text{stuff}) + 16(\text{stuff}) + 64$

Since $(pqr)^2 = 9$ and all other terms are multiples of 4:
\[\text{Answer} \equiv 9 \equiv 1 \pmod{4}\]

Only choice (D) 125 satisfies $125 \equiv 1 \pmod{4}$ $\checkmark$

\begin{warningbox}{Common Pitfalls}

\textbf{Pitfall 1: Complex Number Sign Errors}
\begin{itemize}
    \item \textit{Error}: Computing $i^2 = +1$ instead of $-1$
    \item \textit{Example}: $(10i)^2 = 100i^2 = -100$ (not $+100$)
    \item \textit{Prevention}: Write out $i^2 = -1$ explicitly at each step
\end{itemize}

\textbf{Pitfall 2: Wrong Power of $i$}
\begin{itemize}
    \item \textit{Error}: $(2i)^3 = 8i^3 = 8i$ (wrong!)
    \item \textit{Correct}: $i^3 = i^2 \cdot i = (-1) \cdot i = -i$, so $(2i)^3 = 8i^3 = -8i$
    \item \textit{Prevention}: Remember cycle: $i^1 = i$, $i^2 = -1$, $i^3 = -i$, $i^4 = 1$
\end{itemize}

\textbf{Pitfall 3: Vieta's Formula Sign Errors}
\begin{itemize}
    \item \textit{Error}: For $x^3 + 2x^2 - x + 3$, thinking $p + q + r = 2$ (wrong!)
    \item \textit{Correct}: $p + q + r = -2$ (negative of $x^2$ coefficient)
    \item \textit{Prevention}: Standard form is $x^3 - (p+q+r)x^2 + \ldots$, so sum $= -(\text{coeff of }x^2)$
\end{itemize}

\textbf{Pitfall 4: Difference of Squares Error}
\begin{itemize}
    \item \textit{Error}: $(a + bi)(a - bi) = a^2 - b^2$ (missing the sign)
    \item \textit{Correct}: $(a + bi)(a - bi) = a^2 - (bi)^2 = a^2 - b^2i^2 = a^2 + b^2$
    \item \textit{Example}: $(-5 - 10i)(-5 + 10i) = 25 + 100 = 125$
    \item \textit{Prevention}: Always write $(bi)^2 = b^2i^2 = -b^2$
\end{itemize}

\textbf{Pitfall 5: Algebraic Expansion Arithmetic}
\begin{itemize}
    \item \textit{Error}: When expanding, missing terms or combining incorrectly
    \item \textit{Prevention}: Write out each symmetric function calculation separately and verify
\end{itemize}

\textbf{Pitfall 6: Forgetting to Square in $p^2q^2r^2$}
\begin{itemize}
    \item \textit{Error}: Computing $(pqr)^2 = -3$ instead of $9$
    \item \textit{Correct}: $pqr = -3$, so $(pqr)^2 = (-3)^2 = 9$
    \item \textit{Prevention}: Square the entire product, including sign

\section*{Complete Solution (Detailed)}

\subsection*{Solution Method 1: Complex Number Factorization (Recommended)}

Given polynomial: $f(x) = x^3 + 2x^2 - x + 3$ with roots $p, q, r$.

Find: $(p^2 + 4)(q^2 + 4)(r^2 + 4)$

\textbf{Step 1: Factor using complex numbers}

Recognize that $p^2 + 4 = p^2 - (-4) = p^2 - (2i)^2$ since $i^2 = -1$.

By difference of squares:
\[p^2 + 4 = (p + 2i)(p - 2i)\]

Similarly for $q$ and $r$:
\[q^2 + 4 = (q + 2i)(q - 2i)\]
\[r^2 + 4 = (r + 2i)(r - 2i)\]

\textbf{Step 2: Rewrite the product}

\begin{align*}
(p^2 + 4)(q^2 + 4)(r^2 + 4) &= (p + 2i)(p - 2i)(q + 2i)(q - 2i)(r + 2i)(r - 2i)\\
&= [(p + 2i)(q + 2i)(r + 2i)][(p - 2i)(q - 2i)(r - 2i)]
\end{align*}

\textbf{Step 3: Recognize polynomial evaluation}

Since $f(x) = x^3 + 2x^2 - x + 3$ and $p, q, r$ are its roots:
\[f(x) = (x - p)(x - q)(x - r)\]

Therefore:
\[(p - x)(q - x)(r - x) = (-1)^3(x - p)(x - q)(x - r) = -f(x)\]

With $x = -2i$:
\[(p - (-2i))(q - (-2i))(r - (-2i)) = -f(-2i)\]
\[(p + 2i)(q + 2i)(r + 2i) = -f(-2i)\]

With $x = 2i$:
\[(p - 2i)(q - 2i)(r - 2i) = -f(2i)\]

So our product becomes:
\[[-f(-2i)][-f(2i)] = f(-2i) \cdot f(2i)\]

\textbf{Step 4: Evaluate $f(2i)$}

\begin{align*}
f(2i) &= (2i)^3 + 2(2i)^2 - 2i + 3\\
&= 8i^3 + 2 \cdot 4i^2 - 2i + 3
\end{align*}

Recall: $i^2 = -1$ and $i^3 = i^2 \cdot i = -i$

\begin{align*}
f(2i) &= 8(-i) + 8(-1) - 2i + 3\\
&= -8i - 8 - 2i + 3\\
&= -10i - 5
\end{align*}

\textbf{Step 5: Evaluate $f(-2i)$}

\begin{align*}
f(-2i) &= (-2i)^3 + 2(-2i)^2 - (-2i) + 3\\
&= -8i^3 + 2 \cdot 4i^2 + 2i + 3\\
&= -8(-i) + 8(-1) + 2i + 3\\
&= 8i - 8 + 2i + 3\\
&= 10i - 5
\end{align*}

\textbf{Step 6: Compute the product}

\begin{align*}
f(2i) \cdot f(-2i) &= (-10i - 5)(10i - 5)\\
&= (-5 - 10i)(-5 + 10i)
\end{align*}

This is a difference of squares: $(a - b)(a + b) = a^2 - b^2$ with $a = -5$ and $b = 10i$:

\begin{align*}
&= (-5)^2 - (10i)^2\\
&= 25 - 100i^2\\
&= 25 - 100(-1)\\
&= 25 + 100\\
&= 125
\end{align*}

\textbf{Answer: $\boxed{125}$}

\subsection*{Solution Method 2: Vieta's Formulas (Alternative)}

Given: $f(x) = x^3 + 2x^2 - x + 3$ with roots $p, q, r$

\textbf{Step 1: Extract Vieta's formulas}

For $x^3 + ax^2 + bx + c = 0$ with roots $p, q, r$:
\begin{itemize}
    \item $p + q + r = -a = -2$
    \item $pq + pr + qr = b = -1$
    \item $pqr = -c = -3$
\end{itemize}

\textbf{Step 2: Expand the expression}

\begin{align*}
&(p^2 + 4)(q^2 + 4)(r^2 + 4)\\
&= p^2q^2r^2 + 4(p^2q^2 + p^2r^2 + q^2r^2) + 16(p^2 + q^2 + r^2) + 64
\end{align*}

\textbf{Step 3: Compute $p^2q^2r^2$}

\[p^2q^2r^2 = (pqr)^2 = (-3)^2 = 9\]

\textbf{Step 4: Compute $p^2 + q^2 + r^2$}

\begin{align*}
p^2 + q^2 + r^2 &= (p + q + r)^2 - 2(pq + pr + qr)\\
&= (-2)^2 - 2(-1)\\
&= 4 + 2\\
&= 6
\end{align*}

\textbf{Step 5: Compute $p^2q^2 + p^2r^2 + q^2r^2$}

Square both sides of $pq + pr + qr = -1$:
\begin{align*}
(pq + pr + qr)^2 &= p^2q^2 + p^2r^2 + q^2r^2 + 2(p^2qr + pq^2r + pqr^2)\\
&= p^2q^2 + p^2r^2 + q^2r^2 + 2pqr(p + q + r)
\end{align*}

Substitute known values:
\begin{align*}
(-1)^2 &= p^2q^2 + p^2r^2 + q^2r^2 + 2(-3)(-2)\\
1 &= p^2q^2 + p^2r^2 + q^2r^2 - 12\\
p^2q^2 + p^2r^2 + q^2r^2 &= -11
\end{align*}

\textbf{Step 6: Substitute into expansion}

\begin{align*}
&p^2q^2r^2 + 4(p^2q^2 + p^2r^2 + q^2r^2) + 16(p^2 + q^2 + r^2) + 64\\
&= 9 + 4(-11) + 16(6) + 64\\
&= 9 - 44 + 96 + 64\\
&= 125
\end{align*}

\textbf{Answer: $\boxed{125}$}

\subsection*{Verification: Modular Arithmetic Check}

From the expansion, we have:
\[(p^2 + 4)(q^2 + 4)(r^2 + 4) = (pqr)^2 + 4(\text{terms}) + 16(\text{terms}) + 64\]

Since $(pqr)^2 = 9$ and all other terms are multiples of 4:
\[\text{Answer} \equiv 9 \equiv 1 \pmod{4}\]

Checking answer choices modulo 4:
\begin{itemize}
    \item (A) $64 \equiv 0 \pmod{4}$ $\times$
    \item (B) $75 \equiv 3 \pmod{4}$ $\times$
    \item (C) $100 \equiv 0 \pmod{4}$ $\times$
    \item (D) $125 \equiv 1 \pmod{4}$ $\checkmark$
    \item (E) $144 \equiv 0 \pmod{4}$ $\times$
\end{itemize}

Only (D) 125 satisfies the modular constraint!
